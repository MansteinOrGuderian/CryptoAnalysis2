\begin{enumerate}[label=\alph*)]
    \item $\preceq = \preceq_{\text{lex}}, f (x_{1}, x_{2}, x_{3}) = x_{1} \oplus x_{1}x_{2}x_{3}$;
    \item $\preceq = \preceq_{\text{drl}}, f (x_{1}, x_{2}, x_{3}) = x_{1} \oplus x_{1}x_{2}x_{3}$;
\end{enumerate}
